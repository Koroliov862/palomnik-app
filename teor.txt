ВСЕ СТИЛИ ПИШУТСЯ ЧЕРЕЗ ОБРАЩЕНИЕ К ОБЪЕКТУ
инлайн стили максимум только для картинок
все остальное через StyleSheet.create({

})

Flexbox сетка для контента
по умолчанию весь контент в колонку
flexDirection:"column" - контент в колонку
flexDirection:"row" - контент в стронку

выравнивание элементов по главной оси
justifyContent: 'center'
justifyContent: 'flex-start' - в начало
justifyContent: 'flex-end' - в конец
justifyContent: 'space-between' - расстояние максимальное между элементами
justifyContent: 'space-around' - расстояние вокруг
justifyContent: 'space-evenly' - ранвомерное расстояние

выравнивание элементов по поперечной оси
alignItems: 'center'

само свойства флекс у элемента
flex:1 - означает занять все доступное место
например у главное контейнера в котором другие элементы можно поставить
flex:1
если не поставить флекс 1 контейнер может занимать высоту своего содержимого

import ...

const buttons = [
    {id:'1', title:"первая", role:"дизайн"},
    {id:'1', title:"первая", role:"рпо"},
    {id:'1', title:"первая", role:"бананы"},
    {id:'1', title:"первая", role:"обезьяна"},
]

const List = () => {
    return (
        <FlatList
            data={buttons}
            keyExtractor={(item)=>item.id}
            renderItem={({item}) =>(
                <View>
                     <Text>{item.title}</Text>
                     <Text>{item.role}</Text>
                </View>
            )}
        />
    );
};

export default List;
